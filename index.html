let timerInterval; let timeLeft = 25 * 60; // 25 minutes const timerDisplay = document.getElementById("timer-display"); const startPauseButton = document.getElementById("start-pause-button"); const pomodoroButton = document.getElementById("pomodoro-button"); const shortBreakButton = document.getElementById("short-break-button"); const longBreakButton = document.getElementById("long-break-button"); function updateTimerDisplay() { const minutes = Math.floor(timeLeft / 60); const seconds = timeLeft % 60; timerDisplay.textContent = `${minutes.toString().padStart(2, "0")}:${seconds .toString() .padStart(2, "0")}`; } function startPauseTimer() { if (timerInterval) { clearInterval(timerInterval); timerInterval = null; startPauseButton.textContent = "START"; } else { timerInterval = setInterval(() => { if (timeLeft > 0) { timeLeft--; updateTimerDisplay(); } else { clearInterval(timerInterval); } }, 1000); startPauseButton.textContent = "PAUSE"; } } function setTimer(duration) { clearInterval(timerInterval); timerInterval = null; timeLeft = duration; updateTimerDisplay(); startPauseButton.textContent = "START"; } pomodoroButton.addEventListener("click", () => setTimer(25 * 60)); shortBreakButton.addEventListener("click", () => setTimer(5 * 60)); longBreakButton.addEventListener("click", () => setTimer(15 * 60)); startPauseButton.addEventListener("click", startPauseTimer); // Initialize updateTimerDisplay();
